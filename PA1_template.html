  actdata  # creation of the dataset that will have no more NAs
for (i in 1:nrow(actdata)) {
  if (is.na(actdata$steps[i])) {
    actdata2$steps[i] <- mn_int[[as.character(actdata[i, "interval"])]]
  }
}
